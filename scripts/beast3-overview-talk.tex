\begin{frame}{The Big Picture}
Three pillars of change in BEAST 3:
\vspace{1em}
\begin{enumerate}
\item \textbf{JPMS modules}: replace the custom classloader with the standard Java module system --- gets us to Java 25
\item \textbf{Strongly-typed spec system}: compile-time checked inputs and domain types
\item \textbf{New package distribution}: JPMS service discovery, with Maven Central as the recommended (not required) distribution channel
\end{enumerate}
\vspace{1em}
\begin{block}{Key insight}
JPMS is the \textbf{architecture}. Maven is convenient \textbf{tooling} --- you can build a BEAST 3 package with Gradle, Ant, or anything that produces a modular JAR.
\end{block}
\end{frame}

% ============================================================
\section{Why JPMS (+ Maven)?}
% ============================================================

% ---- Inlined from motivation-slides (condensed for overview) ----

\begin{frame}{Problems We Wanted to Solve}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Classloading \& isolation}
\begin{itemize}
\item Custom classloader predates Java modules by 15+ years
\item Source of ``magic'' that confuses new developers and breaks standard tooling
\item Kept us stuck on old Java versions
\item Using BEAST as a \textbf{library} (e.g.\ LPhyBEAST) was nearly impossible
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Build \& dependencies}
\begin{itemize}
\item Ant \code{build.xml}: manual JAR management, no transitive dependency resolution
\item Adding a library means hand-copying JARs and editing classpaths
\item No standard way to publish or consume packages
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{JPMS is the Architecture, Maven is the Tooling}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{JPMS} (the architecture)
\begin{itemize}
\item Replaces the custom classloader with a standard Java mechanism
\item Strong encapsulation: modules declare exports
\item Per-package \code{ModuleLayer} isolation
\item \code{module-info.java} \code{provides} declarations are the single source of truth for service discovery
\item Gets us to \textbf{Java 25}
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Maven} (recommended tooling)
\begin{itemize}
\item Declared dependencies in \code{pom.xml}; automatic transitive resolution
\item IDE, CI, and Maven Central integration out of the box
\item Reproducible builds
\item \textbf{Not required:} Gradle, Ant, or any tool that produces a modular JAR will work
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\section{Strongly Typed Spec System}
% ============================================================

% ============================================================
% TYPE SAFETY MOTIVATION
% ============================================================

\begin{frame}{The Type Safety Problem}
\textbf{In BEAST 2}, almost everything is \code{Input<RealParameter>} or \code{Input<Function>}:

\vspace{0.5em}
\begin{itemize}
\item Type errors only discovered at \textbf{runtime}: wrong parameter passed to wrong input
\item Bounds (\code{lower="0"}, \code{upper="1"}) are stringly-typed XML attributes
\item No way to express ``this must be a positive real'' in the Java API
\item IDE cannot help you: everything looks like the same type
\end{itemize}

\vspace{1em}
\begin{exampleblock}{Goal}
Make invalid models \textbf{unrepresentable} at compile time.
\end{exampleblock}
\end{frame}

% ============================================================
% SPEC SYSTEM OVERVIEW
% ============================================================

\begin{frame}{Strongly Typed Parameters \& Domains}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Typed parameters}
\begin{itemize}
\item \code{RealScalarParam}: single real
\item \code{RealVectorParam}: vector of reals
\item \code{IntScalarParam} / \code{IntVectorParam}
\item \code{BoolScalarParam} / \code{BoolVectorParam}
\item \code{SimplexParam}: probability simplex
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Domain types}
\begin{itemize}
\item \code{PositiveReal}: replaces \code{lower="0"}
\item \code{UnitInterval}: replaces \code{lower="0" upper="1"}
\item \code{Real}: unbounded
\item \code{NonNegativeInt}: replaces \code{lower="0"}
\item \code{PositiveInt}: replaces \code{lower="1"}
\end{itemize}
\end{column}
\end{columns}

\vspace{1em}
\begin{itemize}
\item Domain is part of the \textbf{Java type}: compiler catches mismatches
\item Legacy classes are \textbf{deprecated, not removed}: migrate incrementally
\end{itemize}
\end{frame}

% BEFORE / AFTER

\begin{frame}[fragile]{Before / After: Input Declaration}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{BEAST 2}
\begin{lstlisting}[language=Java]
public Input<RealParameter> kappaInput =
    new Input<>("kappa",
    "ts/tv ratio",
    Validate.REQUIRED);

// lower="0" only checked at runtime
// in the XML
\end{lstlisting}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{BEAST 3}
\begin{lstlisting}[language=Java]
public Input<RealScalar<PositiveReal>>
    kappaInput = new Input<>("kappa",
    "ts/tv ratio",
    Validate.REQUIRED);

// PositiveReal is in the type!
// Compiler rejects wrong domain
\end{lstlisting}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Package System Overview}
\begin{itemize}
\item \textbf{One ModuleLayer per package:} namespace isolation without losing visibility to the core API
\item \textbf{Same code path for dev and deployment:} \code{module-info.java} \code{provides} declarations are the single source of truth
\item \textbf{Graceful degradation:} modules with unsatisfied dependencies are silently skipped
\end{itemize}

% TWO DISTRIBUTION MODELS
% ============================================================

\begin{frame}{Package Distribution: Maven Central \& ZIP}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Maven packages} (recommended)
\begin{itemize}
\item Published to Maven Central
\item Install via BEAUti or CLI
\item Automatic dependency resolution
\item \code{maven-packages.xml} tracks installs
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{ZIP packages} (legacy, still works)
\begin{itemize}
\item Traditional CBAN archives
\item One subdirectory per package
\item \code{version.xml} for service discovery
\item Self-describing directory structure
\end{itemize}
\end{column}
\end{columns}

\vspace{1em}
\begin{block}{}
Both formats coexist and both use the \textbf{same JPMS \code{ModuleLayer} loader} at runtime. Loading precedence: boot layer $\to$ Maven $\to$ ZIP.
\end{block}
\end{frame}

\section{Migrating Your Package}
% ============================================================

% ============================================================
% STEP 1: BUILD
% ============================================================

\begin{frame}[fragile]{Step 1: Set Up Your Build (Maven Recommended)}
Replace \code{build.xml} with a \code{pom.xml} (or use Gradle --- any tool that produces a modular JAR):

\begin{lstlisting}[language=XML]
<project>
  <groupId>io.github.yourorg</groupId>
  <artifactId>beast-mypackage</artifactId>
  <version>1.0.0-SNAPSHOT</version>

  <dependencies>
    <dependency>
      <groupId>io.github.compevol</groupId>
      <artifactId>beast-base</artifactId>
      <version>2.8.0-SNAPSHOT</version>
    </dependency>
  </dependencies>
</project>
\end{lstlisting}

\begin{itemize}
\item Requires JDK 25+ and Maven 3.9+
\item Build BEAST 3 locally first: \code{mvn install -DskipTests}
\item See \code{beast-package-skeleton} for a complete template POM
\end{itemize}
\end{frame}

% ============================================================
% STEP 2: MODULE-INFO
% ============================================================

\begin{frame}[fragile]{Step 2: Add \code{module-info.java}}
Create \code{src/main/java/module-info.java}:

\begin{lstlisting}[language=Java]
open module my.beast.package {
    requires beast.pkgmgmt;
    requires beast.base;
    requires static beast.fx;        // optional at runtime
    requires static javafx.controls; // optional at runtime

    exports my.beast.package;

    provides beast.base.core.BEASTInterface with
        my.beast.package.MyModel,
        my.beast.package.MyOperator,
        my.beast.package.app.beauti.MyInputEditor;
}
\end{lstlisting}

\begin{itemize}
\item Every \code{BEASTInterface} subclass \textbf{must} be listed in \code{provides}
\item \code{requires static}: dependency is optional at runtime (headless OK)
\item GUI classes go in \code{*.app.beauti} subpackage by convention
\end{itemize}
\end{frame}

% ============================================================
% STEP 3: API CHANGES
% ============================================================

\begin{frame}{Step 3: Fix Compile Errors}
Common changes needed when migrating package code:

\vspace{0.5em}
\begin{center}
\small
\begin{tabular}{ll}
\toprule
\textbf{BEAST 2 pattern} & \textbf{BEAST 3 replacement} \\
\midrule
\code{StateNode} implements \code{Function} & No longer a \code{Function}; \\
& use spec types (\code{RealScalar}, etc.) \\
\code{stateNode.scale(s)} & Implement \code{Scalable} interface \\
\code{Prior} + distribution wrapper & Distribution is the prior directly \\
\code{Input<RealParameter>} & \code{Input<RealScalar<PositiveReal>{}>} \\
\code{Evaluator} & Removed; refactor MCMC usage \\
\bottomrule
\end{tabular}
\end{center}

\vspace{0.5em}
\begin{block}{}
Legacy classes (\code{RealParameter}, \code{Prior}, \code{Function}) are \textbf{deprecated, not removed}. Your package will still compile --- migrate to spec types incrementally.
\end{block}
\end{frame}

% ============================================================
% STEP 4: TEST & PUBLISH
% ============================================================

\begin{frame}{Step 4: Test \& Publish}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Testing}
\begin{itemize}
\item \code{mvn test} runs JUnit tests
\item IntelliJ auto-discovers JPMS modules --- run/debug XML from IDE
\item Namespace test resources under your module to avoid split-package errors
\end{itemize}

\vspace{0.5em}
\textbf{Publishing to Maven Central}
\begin{itemize}
\item Add release profile to POM (see skeleton)
\item Tag release $\to$ GitHub Actions auto-publishes JAR
\item Users install via BEAUti with automatic dependency resolution
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Quick-start checklist}
\begin{enumerate}
\item Clone \code{beast-package-skeleton}
\item Rename module, packages, POM coords
\item Copy your source into \code{src/main/java/}
\item \code{mvn compile} --- fix errors
\item Add classes to \code{module-info.java} \code{provides}
\item \code{mvn test}
\item Update BEAST XML specs if needed
\end{enumerate}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Key Breaking Changes}
\begin{center}
\begin{tabular}{lll}
\toprule
\textbf{Area} & \textbf{BEAST 2} & \textbf{BEAST 3} \\
\midrule
Modules & Custom classloader & JPMS \code{ModuleLayer} \\
Build & Ant, Java 8/11 & Maven (recommended), Java 25 \\
Parameters & \code{RealParameter} & Typed spec params + domains \\
Priors & \code{Prior} wrapper & Distribution is the prior \\
\code{StateNode} & implements \code{Function} & No longer a \code{Function} \\
\code{StateNode.scale()} & built-in & Removed; use \code{Scalable} \\
JavaFX & Bundled with JDK & Maven dependency \\
\bottomrule
\end{tabular}
\end{center}

\vspace{0.5em}
Legacy classes are \textbf{deprecated, not removed}: migrate incrementally.
\end{frame}
